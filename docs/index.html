Summary

Helper Description

gt Greater than >.

{{opr}}

Compare two values using operators.

Params:

  • any val1 First value to be compared with second.
  • '=='|'==='|'!='|'!=='|'<'|'>'|'<='|'>=' opr The operator to use. Operators must be enclosed in quotes: `">"`, `"="`, `"<="`, and so on.
  • any val2 Second value to be compared with first.

Example:

// Output: false
hbs.compile("{{opr a '===' b}}")({"a":"3","b":3});

// Output: false
hbs.compile("{{#if (opr a '===' b)}}true{{else}}false{{/if}}")({"a":"3","b":3});

// Output: true
hbs.compile("{{opr a '==' b}}")({"a":"3","b":3});

// Output: true
hbs.compile("{{#if (opr a '==' b)}}true{{else}}false{{/if}}")({"a":"3","b":3});

// Output: true
hbs.compile("{{opr a '!==' b}}")({"a":4,"b":3});

// Output: true
hbs.compile("{{#if (opr a '!==' b)}}true{{else}}false{{/if}}")({"a":4,"b":3});

// Output: false
hbs.compile("{{opr a '!=' b}}")({"a":"3","b":3});

// Output: false
hbs.compile("{{#if (opr a '!=' b)}}true{{else}}false{{/if}}")({"a":"3","b":3});

// Output: true
hbs.compile("{{opr a '<' b}}")({"a":2,"b":3});

// Output: true
hbs.compile("{{#if (opr a '<' b)}}true{{else}}false{{/if}}")({"a":2,"b":3});

// Output: false
hbs.compile("{{opr a '>' b}}")({"a":2,"b":3});

// Output: false
hbs.compile("{{#if (opr a '>' b)}}true{{else}}false{{/if}}")({"a":2,"b":3});

// Output: true
hbs.compile("{{opr a '<=' b}}")({"a":2,"b":3});

// Output: true
hbs.compile("{{#if (opr a '<=' b)}}true{{else}}false{{/if}}")({"a":2,"b":3});

// Output: true
hbs.compile("{{opr a '>=' b}}")({"a":3,"b":3});

// Output: true
hbs.compile("{{#if (opr a '>=' b)}}true{{else}}false{{/if}}")({"a":3,"b":3});

{{eq}}

Determine whether or not two values are equal (===).

Params:

  • any val1 First value to be compared with second.
  • any val2 Second value to be compared with first.

Example:

// Output: false
hbs.compile("{{eq a b}}")({"a":"3","b":3});

// Output: false
hbs.compile("{{#if (eq a b)}}true{{else}}false{{/if}}")({"a":"3","b":3});

{{eqw}}

Params:

  • any val1 First value to be compared with second.
  • any val2 Second value to be compared with first.

Example:

// Output: true
hbs.compile("{{eqw a b}}")({"a":"3","b":3});

// Output: true
hbs.compile("{{#if (eqw a b)}}true{{else}}false{{/if}}")({"a":"3","b":3});

// Output: true

// Output: true
hbs.compile("{{#if (neq a b)}}true{{else}}false{{/if}}")({"a":4,"b":3});

{{ifx}}

Helper to imitate the ternary '?:' conditional operator.

Params:

  • any val1 Value to return when the condition holds true.
  • any val2 Value to return when the condition is false (Optional).

Example:

// Output: foo
hbs.compile("{{ifx true a b}}")({"a":"foo","b":"bar"});

// Output: bar
hbs.compile("{{ifx false a b}}")({"a":"foo","b":"bar"});

{{not}}

Logical NOT of any expression.

Params:

  • any expr Any expression.

Example:

// Output: false
hbs.compile("{{not true}}")();

// Output: false
hbs.compile("{{#if (not true)}}true{{else}}false{{/if}}")();

// Output: true
hbs.compile("{{not false}}")();

// Output: true
hbs.compile("{{#if (not false)}}true{{else}}false{{/if}}")();

{{empty}}

Check if an array is empty.

Params:

  • any[] coll Array/object to be checked.

Example:

// Output: true
hbs.compile("{{empty coll}}")({"coll":[]});

// Output: true
hbs.compile("{{#if (empty coll)}}true{{else}}false{{/if}}")({"coll":[]});

// Output: false
hbs.compile("{{empty coll}}")({"coll":["foo"]});

// Output: false
hbs.compile("{{#if (empty coll)}}true{{else}}false{{/if}}")({"coll":["foo"]});

{{count}}

Determine the length of an array.

Params:

  • any[] coll Array whose elements to be counted.

Example:

// Output: 2
hbs.compile("{{count coll}}")({"coll":["foo","bar"]});

{{and}}

Returns the boolean AND of two or more parameters passed i.e it is true iff all the parameters are true.

Params:

  • ...any args Any number of boolean parameters.

Example:

// Output: true
hbs.compile("{{and a b}}")({"a":true,"b":true});

// Output: true
hbs.compile("{{#if (and a b)}}true{{else}}false{{/if}}")({"a":true,"b":true});

hbs.compile("{{and a b}}")({"a":false,"b":true});

// Output: false
hbs.compile("{{#if (and a b)}}true{{else}}false{{/if}}")({"a":false,"b":true});

{{coalesce}}

Returns the first non-falsy value from the parameter list.
Checks for the first non-false parameter.

Params:

  • ...any args Any number of parameters.

Example:

// Output: foo
hbs.compile("{{coalesce a b c}}")({"a":"foo","b":"bar","c":"baz"});

// Output: bar
hbs.compile("{{coalesce a b c}}")({"a":"","b":"bar","c":"baz"});

{{includes}}

Returns true if the array contains a particular value, false if it does not.

Params:

  • any[] coll The array.
  • any val The value to search for.
  • boolean strict FALSE for non-strict checking. TRUE by default.

Example:

// Output: true
hbs.compile("{{includes coll a}}")({"coll":[1,2,3,4],"a":2});

// Output: true
hbs.compile("{{#if (includes coll a)}}true{{else}}false{{/if}}")({"coll":[1,2,3,4],"a":2});

// Output: false
hbs.compile("{{includes coll a}}")({"coll":[1,2,3,4],"a":10});

// Output: false
hbs.compile("{{#if (includes coll a)}}true{{else}}false{{/if}}")({"coll":[1,2,3,4],"a":10});

// Output: false
hbs.compile("{{includes coll a}}")({"coll":[1,2,3,4],"a":"3"});

// Output: false
hbs.compile("{{#if (includes coll a)}}true{{else}}false{{/if}}")({"coll":[1,2,3,4],"a":"3"});

// Output: true
hbs.compile("{{includes coll a strict}}")({"coll":[1,2,3,4],"a":"3","strict":false});

// Output: true
hbs.compile("{{#if (includes coll a strict)}}true{{else}}false{{/if}}")({"coll":[1,2,3,4],"a":"3","strict":false});

Date helper

Summary

Helper Description
formatDate Use moment to format the date.

{{formatDate}}

Use moment to format the date.

Params:

  • string format A format string based on moment.
  • string date Date string to format.
  • string|string[] locale Language or language-country locale string (or array of strings) available in https://github.com/moment/moment/tree/develop/locale.

Example:

// Output: 2021/10/25
hbs.compile("{{formatDate 'YYYY/MM/DD' date}}")({"date":"2021-10-24T23:11:14.275Z"});

// Output: 2021/10/25
hbs.compile("{{formatDate 'YYYY/MM/DD' date 'jp'}}")({"date":"2021-10-24T23:11:14.275Z"});

// Output: 2021/10/25
hbs.compile("{{formatDate 'YYYY/MM/DD' date 'es'}}")({"date":"2021-10-24T23:11:14.275Z"});

HTML helper

Summary

Helper Description
classIf Set the CSS classes if the condition is true.

{{classIf}}

Set the CSS classes if the condition is true.

Params:

  • boolean expr Condition to be checked.
  • string cls CSS class to set if the condition is true.

Example:

// Output: foo
hbs.compile("{{classIf expr 'foo'}}")({"expr":true});

{{selectedIf}}

Sets the selected attribute if the condition is true.

Params:

  • boolean expr Condition to be checked.

Example:

// Output: selected
hbs.compile("{{selectedIf expr}}")({"expr":true});

{{checkedIf}}

Sets the checked attribute if the condition is true.

Params:

  • boolean expr Condition to be checked.

Example:

// Output: checked
hbs.compile("{{checkedIf expr}}")({"expr":true});

{{options}}

Generates a select drop-down option list.

Params:

  • {[key: string]: string}[] data List of data.
  • {value: string, text: string} opts Key names for the selected and displayed values of option in the data element (Optional).

Example:

// Output: <option value="1">foo</option>
//         <option value="2" selected>bar</option>
hbs.compile("{{{options data selected='2'}}}")({"data":[{"value":1,"text":"foo"},{"value":2,"text":"bar"}]});

// Output: <option value="392" selected>JAPAN</option>
//         <option value="840">UNITED STATES</option>
hbs.compile("{{{options data selected='392' value='code' text='name'}}}")({"data":[{"code":392,"name":"JAPAN"},{"code":840,"name":"UNITED STATES"}]});

String helper

Summary

Helper Description
slice Extract a portion of a string.
nltobr Replace \n with <br> tags.

It uses sprintf-js internally.
Check https://github.com/alexei/sprintf.js for more information.

{{slice}}

Extract a portion of a string.

Params:

  • string val Target string.
  • number beginIdx The zero-based index at which to begin extraction.
  • number endIdx The zero-based index before which to end extraction. The character at this index will not be included.

Example:

// Output: Just
hbs.compile("{{slice str 0 4}}")({"str":"Just Wow"});

// Output: Wow
hbs.compile("{{slice str 5}}")({"str":"Just Wow"});

// Output:  Wow
hbs.compile("{{slice str -4}}")({"str":"Just Wow"});

{{nltobr}}

Replace \n with <br> tags.

Params:

  • string val Target string.

Example:

// Output: It's<br>just<br>now
hbs.compile("{{{nltobr str}}}")({"str":"It's\njust\nnow"});

{{sprintf}}

Returns a string produced according to the formatting string format.
It uses sprintf-js internally.
Check https://github.com/alexei/sprintf.js for more information.

Params:

  • string format Format string.
  • ...any args Any number of parameters/values.

Example:

// Argument swapping:
// Output: Hello Dolly!
hbs.compile("{{sprintf '%s %s!' str1 str2}}")({"str1":"Hello","str2":"Dolly"});

// Output: foo bar 55 baz 20
hbs.compile("{{sprintf '%s %s %d %s %d' 'foo' 'bar' 55 'baz' '20'}}")();

// Named arguments:
// Output: Hello Dolly
hbs.compile("{{sprintf 'Hello %(name)s' user}}")({"user":{"name":"Dolly"}});

// Output: Hello Dolly
hbs.compile("{{sprintf 'Hello %(name)s' name=str}}")({"str":"Dolly"});

{{join}}

Join the elements of an array using a delimeter.

Params:

  • string[] coll An array of elements to be joined.
  • string delim A delimiter that joins array elements.

Example:

// Output: Hands & legs & feet
hbs.compile("{{{join coll ' & '}}}")({"coll":["Hands","legs","feet"]});
